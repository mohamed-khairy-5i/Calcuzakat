# فحص جودة تلقائي: يعمل عند كل رفع (push) وكل طلب دمج (pull request)
# على الفرع الرئيسي. يتحقق من الأنواع، الاختبارات، والبناء الناجح.
# ملاحظة: هذا الفحص لا يرفع أي كود ولا يُحسب كمساهم في المستودع.
name: CI

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]
  workflow_dispatch:

permissions:
  contents: read

jobs:
  quality:
    name: فحص الأنواع والاختبارات والبناء
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Node.js 20
        uses: actions/setup-node@v4
        with:
          node-version: "20"
          cache: "npm"

      - name: Install dependencies
        run: npm ci

      - name: Type check (tsc --noEmit)
        run: npm run typecheck

      - name: Unit tests (Vitest)
        run: npm test

      - name: Production build (Astro)
        run: npm run build
